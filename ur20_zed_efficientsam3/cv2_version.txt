#!/usr/bin/env python3
import rclpy
from rclpy.node import Node
from sensor_msgs.msg import Image
from cv_bridge import CvBridge
import cv2
import matplotlib as plt
import numpy as np

class UserInput(Node): 
    def __init__(self):
        super().__init__("user_input") 
        self.bridge_ = CvBridge()
        self.image_subscriber_ = self.create_subscription(Image, "/zedx/zed_node/rgb/image_rect_color", self.image_callback, 10)
        self.latest_frame_ = None
        #self.input_timer_ = self.create_timer(0.5, self.input_timer_callback)
        self.get_logger().info("User Input node initialized!")
        #self.get_logger().info("Press ` to take a snapshot.")

def image_callback(self, msg):
    try:
        cv_image = self.bridge_.imgmsg_to_cv2(msg, desired_encoding="bgr8")
        self.latest_frame_ = cv_image
    except:
        self.get_logger().info("Image conversion to cv2 format failed.")

def main(args=None):
    rclpy.init(args=args)
    node = UserInput() 
    rclpy.spin(node)
    rclpy.shutdown()

if __name__ == "__main__":
    main()
